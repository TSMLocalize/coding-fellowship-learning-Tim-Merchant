/* COMPLEX SELECTORS */ /* descendent selector - item within item */ /* selects grandchildren etc. */ article ul { } /* direct descendent selector */ /* only selects the direct descendent */ article > ul { } /* sibling selector */ /* find me all the next articles that are at the same level */ article ~ article { } /* adjacent sibling selector */ /* only selects the next sibling */ article + article { } /* select by attribute */ input[disabled] { } /* target something by its attribute and value */ input[href='design.pdf']{ } input[type="submit"]{ } /* '*' = Look for ".doc" as part of any attribute */ a[href*=".doc"]{ } a[href$=".pdf"]{ } /* pseudo before element */ /* can be used for tool-tips */ input:checked::before { } a:hover { } /* while selected or tabbed into */ input:focus { } /* selects child */ nav > ul > li:nth-child(1){ } nav:first-child { } nav:last-child { } # */ section:target {} /* no children */ div:empty {} /* does not have */ article:not(.about) {} ``` --- ### Pseudo elements ```css div::before section::after p::first-letter p::first-line p::selection ``` ---
